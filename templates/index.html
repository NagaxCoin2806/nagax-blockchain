import requests import hashlib import json import time # URL do seu servidor hospedado no Render URL_REDE = "https://nagax-blockchain.onrender.com" def ordenar_json_stringify(obj): # Ordena as chaves do dicionário para garantir que o hash bata com o validador Python do servidor return json.dumps(obj, sort_keys=True, separators=(',', ':')) def minerar(endereco_minerador): if not endereco_minerador: print("❌ Erro: Você precisa definir um endereço público para receber as recompensas!") return print("🚀 Minerador Nagax Network iniciado...") while True: try: # 1. Busca o estado atual da rede (Cadeia, Mempool e Recompensa) resposta = requests.get(f"{URL_REDE}/chain") if resposta.status_code != 200: print("❌ Falha ao obter dados da blockchain. Tentando novamente...") time.sleep(4) continue dados_rede = resposta.json() cadeia = dados_rede["cadeia"] mempool = dados_rede.get("mempool", []) recompensa_atual = dados_rede["recompensa_bloco_atual"] ultimo_bloco = cadeia[-1] indice = ultimo_bloco["indice"] + 1 hash_anterior = ultimo_bloco["hash"] dificuldade = ultimo_bloco["dificuldade"] quantidade_txs = len(mempool) # Exibe o painel de informações atualizado no terminal print(f"⛏️ Minerando Bloco #{indice} | Dificuldade: {dificuldade} | Mempool: {quantidade_txs} txs") # 2. Monta a transação Coinbase (Recompensa do Minerador) + Taxas total_taxas = 0.0 transacoes_tratadas = json.loads(json.dumps(mempool)) # Clone profundo da mempool for tx in transacoes_tratadas: total_taxas += float(tx.get("taxa", 0.0)) tx["minerador_taxa"] = endereco_minerador tx_recompensa = { "remetente": "REDE (Coinbase)", "destinatario": endereco_minerador, "quantidade": round(recompensa_atual + total_taxas, 4), "taxa": 0.0, "info_recompensa": f"Base: {recompensa_atual} + Taxas: {total_taxas}", "assinatura": "COINBASE_SIGN" } # Junta a recompensa com as demais transações da fila bloco_transacoes = [tx_recompensa] + transacoes_tratadas # Normaliza os campos para evitar divergência de tipos (String vs Float) bloco_transacoes_normalizado = [] for tx in bloco_transacoes: bloco_transacoes_normalizado.append({ "assinatura": tx["assinatura"], "destinatario": tx["destinatario"], "info_recompensa": tx.get("info_recompensa", ""), "minerador_taxa": tx.get("minerador_taxa", ""), "quantidade": float(tx["quantidade"]), "remetente": tx["remetente"], "taxa": float(tx.get("taxa", 0.0)) }) transacoes_ordenadas_str = ordenar_json_stringify(bloco_transacoes_normalizado) timestamp = int(time.time()) alvo = "0" * dificuldade nonce = 0 bloco_minerado = False tempo_inicio = time.time() # 3. Loop de Criptografia (Proof of Work) while True: conteudo = f"{indice}{hash_anterior}{transacoes_ordenadas_str}{timestamp}{nonce}" hash_tentativa = hashlib.sha256(conteudo.encode()).hexdigest() if hash_tentativa.startswith(alvo): tempo_gasto = time.time() - tempo_inicio hash_rate = nonce / tempo_gasto if tempo_gasto > 0 else nonce print(f"🎉 Bloco #{indice} MINERADO! Nonce: {nonce} ({hash_rate/1000:.2f} KH/s)") bloco_pronto = { "indice": indice, "hash_anterior": hash_anterior, "transacoes": bloco_transacoes_normalizado, "timestamp": timestamp, "nonce": nonce, "dificuldade": dificuldade, "hash": hash_tentativa } # 4. Envia o bloco para validação e persistência do nó oficial print("📤 Enviando bloco minerado para o Render...") envio = requests.post(f"{URL_REDE}/blocks/receive", json=bloco_pronto) if envio.status_code == 200: res_dados = envio.json() if "mensagem" in res_dados: print(f"✅ Bloco aceito e adicionado à blockchain oficial!") else: print(f"❌ Rejeitado pelo nó: {res_dados.get('erro')}") else: print(f"❌ Erro crítico de rede HTTP {envio.status_code}") bloco_minerado = True break nonce += 1 # A cada 50.000 tentativas, verifica se a rede mudou ou se novos blocos entraram if nonce % 50000 == 0: break # Pequena pausa tática para aliviar o processamento antes do próximo bloco time.sleep(2) except Exception as e: print(f"❌ Ocorreu uma interrupção ou falha de conexão: {e}") time.sleep(5) if __name__ == "__main__": # COLE AQUI o seu endereço de carteira gerado pelo index.html para coletar os blocos SUA_CARTEIRA_PUBLICA = "INSIRA_SEU_ENDERECO_PUBLICO_AQUI" if SUA_CARTEIRA_PUBLICA == https://nagax-blockchain.onrender.com: print("⚠️ Aviso: Altere a variável 'SUA_CARTEIRA_PUBLICA' com sua chave pública antes de minerar.") else: minerar(SUA_CARTEIRA_PUBLICA)
